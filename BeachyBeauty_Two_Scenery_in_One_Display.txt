//Sun animation
GLfloat position = 0.0f;
GLfloat speed = 0.02f;

//Left cloud
GLfloat position1 = 0.0f;
GLfloat speed1 = 0.06f;

//Right cloud
GLfloat position2 = 0.0f;
GLfloat speed2 = 0.04f;

//Boat animation
GLfloat boatposition = 0.0f;
GLfloat boatspeed = 0.05f;


//Scenery Position
GLfloat sceneryPos1 = 0.0f;
GLfloat scenerySpeed1 = 0.025f;
GLfloat sceneryPos2 = 2.0f;
GLfloat scenerySpeed2 = 0.025f;

void update(int value)
{

//Sun
    if(position <=0.0)
    {
        position = 1.0f;
    }
    position -= speed;
    //----------------

//Left cloud
    if(position1 >2.0)
    {
        position1 = -2.0f;
    }
    position1 += speed1;
    //----------------

    //Right cloud
    if(position2 <-2.0)
    {
        position2 = 2.0f;
    }
    position2 -= speed2;
    //-------------

    glutPostRedisplay();

    //Sboat
    if(boatposition >1)
    {
        boatposition = -1.0f;
    }
    boatposition += boatspeed;

    //Scenery 1
    if(sceneryPos1 >2)
    {
        sceneryPos1 = -2.0f+scenerySpeed2;
    }
    sceneryPos1 += scenerySpeed1;

    //Scenery 2
    if(sceneryPos2 >2)
    {
        sceneryPos2 = -2.0f+scenerySpeed1;
    }
    sceneryPos2 += scenerySpeed2;

    //update screen
    glutTimerFunc(100, update, 0);
}

//**************Keypress effect**************
void handleKeypress(unsigned char key, int x, int y)
{
    switch (key)
    {
    case 's':
        speed = 0.0f;
        break;
    case 'r':
        speed = 0.1f;
        break;
    case 'f':
        speed += 0.1f;
        break;
    case 'd':
        speed -= 0.1f;
        break;

    //Cloud left
    case 'y':
        speed1 = 0.0f;
        break;
    case 'z':
        speed1 += 0.1f;
        break;


        glutPostRedisplay();
    }



}

//*****************MousePress********************
void handleMouse(int button, int state, int x, int y)
{
    if (button == GLUT_LEFT_BUTTON)
    {
        speed += 0.02f;
    }
    if (button == GLUT_MIDDLE_BUTTON)
    {
        speed = 0.0f;
    }
    if (button == GLUT_RIGHT_BUTTON)
    {
        speed -= 0.02f;
    }
    glutPostRedisplay();
}



int main(int argc, char** argv)
{
    glutInit(&argc, argv);
    glutCreateWindow("Circle");
    glutInitWindowSize(720, 480);
    glutDisplayFunc(display);
    glutTimerFunc(100, update, 0);
    glutKeyboardFunc(handleKeypress);
    glutMouseFunc(handleMouse);
    glutMainLoop();
    return 0;
}
